642  inline void SColorHSL::fromRGB(const SColorf &color)
643  {
644  const f32 maxVal = core::max_(color.getRed(), color.getGreen(), color.getBlue());
645  const f32 minVal = (f32)core::min_(color.getRed(), color.getGreen(), color.getBlue());
646  Luminance = (maxVal+minVal)*50;
647  if (core::equals(maxVal, minVal))
648  {
649  Hue=0.f;
650  Saturation=0.f;
651  return;
652  }
653 
654  const f32 delta = maxVal-minVal;
655  if ( Luminance <= 50 )
656  {
657  Saturation = (delta)/(maxVal+minVal);
658  }
659  else
660  {
661  Saturation = (delta)/(2-maxVal-minVal);
662  }
663  Saturation *= 100;
664 
665  if (core::equals(maxVal, color.getRed()))
666  Hue = (color.getGreen()-color.getBlue())/delta;
667  else if (core::equals(maxVal, color.getGreen()))
668  Hue = 2+((color.getBlue()-color.getRed())/delta);
669  else // blue is max
670  Hue = 4+((color.getRed()-color.getGreen())/delta);
671 
672  Hue *= 60.0f;
673  while ( Hue < 0.f )
674  Hue += 360;
675  }

678  inline void SColorHSL::toRGB(SColorf &color) const
679  {
680  const f32 l = Luminance/100;
681  if (core::iszero(Saturation)) // grey
682  {
683  color.set(l, l, l);
684  return;
685  }
686 
687  f32 rm2;
688 
689  if ( Luminance <= 50 )
690  {
691  rm2 = l + l * (Saturation/100);
692  }
693  else
694  {
695  rm2 = l + (1 - l) * (Saturation/100);
696  }
697 
698  const f32 rm1 = 2.0f * l - rm2;
699 
700  const f32 h = Hue / 360.0f;
701  color.set( toRGB1(rm1, rm2, h + 1.f/3.f),
702  toRGB1(rm1, rm2, h),
703  toRGB1(rm1, rm2, h - 1.f/3.f)
704  );
705  }

708  // algorithm from Foley/Van-Dam
709  inline f32 SColorHSL::toRGB1(f32 rm1, f32 rm2, f32 rh) const
710  {
711  if (rh<0)
712  rh += 1;
713  if (rh>1)
714  rh -= 1;
715 
716  if (rh < 1.f/6.f)
717  rm1 = rm1 + (rm2 - rm1) * rh*6.f;
718  else if (rh < 0.5f)
719  rm1 = rm2;
720  else if (rh < 2.f/3.f)
721  rm1 = rm1 + (rm2 - rm1) * ((2.f/3.f)-rh)*6.f;
722 
723  return rm1;
724  }
