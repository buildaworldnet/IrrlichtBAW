427  static void OnAnimate_static(IDummyTransformationSceneNode* node, u32 timeMs)
428  {
429  ISceneNode* tmp = static_cast<ISceneNode*>(node);
430  if (!node->isISceneNode()||tmp->IsVisible)
431  {
432  // animate this node with all animators
433 
434  ISceneNodeAnimatorList::ConstIterator ait = node->getAnimators().begin();
435  while (ait != node->getAnimators().end())
436  {
437  // continue to the next node before calling animateNode()
438  // so that the animator may remove itself from the scene
439  // node without the iterator becoming invalid
440  ISceneNodeAnimator* anim = *ait;
441  ++ait;
442  anim->animateNode(node, timeMs);
443  }
444 
445  // update absolute position
446  node->updateAbsolutePosition();
447 
448  // perform the post render process on all children
449 
450  IDummyTransformationSceneNodeList::ConstIterator it = node->getChildren().begin();
451  for (; it != node->getChildren().end(); ++it)
452  {
453  ISceneNode* tmpChild = static_cast<ISceneNode*>(*it);
454  if ((*it)->isISceneNode())
455  static_cast<ISceneNode*>(*it)->OnAnimate(timeMs);
456  else
457  OnAnimate_static(*it,timeMs);
458  }
459  }
460  }

462  static void OnRegisterSceneNode_static(IDummyTransformationSceneNode* node)
463  {
464  ISceneNode* tmp = static_cast<ISceneNode*>(node);
465  if (!node->isISceneNode()||tmp->IsVisible)
466  {
467  IDummyTransformationSceneNodeList::ConstIterator it = node->getChildren().begin();
468  for (; it != node->getChildren().end(); ++it)
469  {
470  if ((*it)->isISceneNode())
471  static_cast<ISceneNode*>(*it)->OnRegisterSceneNode();
472  else
473  OnRegisterSceneNode_static(*it);
474  }
475  }
