176  inline virtual size_t needsDeepAbsoluteTransformRecompute() const
177  {
178  const IDummyTransformationSceneNode* parentStack[1024];
179  parentStack[0] = this;
180  size_t stackSize=0;
181 
182  while (parentStack[stackSize])
183  {
184  parentStack[++stackSize] = parentStack[stackSize];
185  if (stackSize>=1024)
186  break;
187  }
188 
189  size_t maxStackSize = stackSize-1;
190  while (--stackSize)
191  {
192  if (parentStack[stackSize]->relativeTransNeedsUpdate||parentStack[stackSize]->lastTimeRelativeTransRead[3]<parentStack[stackSize]->relativeTransChanged)
193  return stackSize;
194 
195  if (stackSize<maxStackSize)
196  {
197  if (parentStack[stackSize]->lastTimeRelativeTransRead[4]<parentStack[stackSize+1]->getAbsoluteTransformLastRecomputeHint())
198  return stackSize;
199  }

202  return 0xdeadbeefu;

228  {
229  bool recompute = relativeTransNeedsUpdate||lastTimeRelativeTransRead[3]<relativeTransChanged;
230 
231  if (Parent)

233  uint64_t parentAbsoluteHint = Parent->getAbsoluteTransformLastRecomputeHint();
234  if (lastTimeRelativeTransRead[4] < parentAbsoluteHint)

245  lastTimeRelativeTransRead[3] = relativeTransChanged;
246  }
247  }

251  lastTimeRelativeTransRead[3] = relativeTransChanged;
