202  //http://momchil-velikov.blogspot.fr/2013/10/fast-sse-quternion-multiplication.html
203  inline quaternion operator*(const quaternion& other) const
204  {
205  __m128 xyzw = vectorSIMDf::getAsRegister();
206  __m128 abcd = reinterpret_cast<const vectorSIMDf&>(other).getAsRegister();
207 
208  __m128 t0 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(abcd), _MM_SHUFFLE (3, 3, 3, 3))); /* 1, 0.5 */
209  __m128 t1 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(xyzw), _MM_SHUFFLE (2, 3, 0, 1))); /* 1, 0.5 */
210 
211  __m128 t3 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(abcd), _MM_SHUFFLE (0, 0, 0, 0))); /* 1, 0.5 */
212  __m128 t4 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(xyzw), _MM_SHUFFLE (1, 0, 3, 2))); /* 1, 0.5 */
213 
214  __m128 t5 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(abcd), _MM_SHUFFLE (1, 1, 1, 1))); /* 1, 0.5 */
215  __m128 t6 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(xyzw), _MM_SHUFFLE (2, 0, 3, 1))); /* 1, 0.5 */
216 
217  /* [d,d,d,d]*[z,w,x,y] = [dz,dw,dx,dy] */
218  __m128 m0 = _mm_mul_ps (t0, t1); /* 5/4, 1 */
219 
220  /* [a,a,a,a]*[y,x,w,z] = [ay,ax,aw,az]*/
221  __m128 m1 = _mm_mul_ps (t3, t4); /* 5/4, 1 */
222 
223  /* [b,b,b,b]*[z,x,w,y] = [bz,bx,bw,by]*/
224  __m128 m2 = _mm_mul_ps (t5, t6); /* 5/4, 1 */
225 
226  /* [c,c,c,c]*[w,z,x,y] = [cw,cz,cx,cy] */
227  __m128 t7 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(abcd), _MM_SHUFFLE (2, 2, 2, 2))); /* 1, 0.5 */
228  __m128 t8 = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(xyzw), _MM_SHUFFLE (3, 2, 0, 1))); /* 1, 0.5 */
229 
230  __m128 m3 = _mm_mul_ps (t7, t8); /* 5/4, 1 */
231 
232  /* 1 */
233  /* [dz,dw,dx,dy]+-[ay,ax,aw,az] = [dz+ay,dw-ax,dx+aw,dy-az] */
234  __m128 e = _mm_addsub_ps (m0, m1); /* 3, 1 */
235 
236  /* 2 */
237  /* [dx+aw,dz+ay,dy-az,dw-ax] */
238  e = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(e), _MM_SHUFFLE (1, 3, 0, 2))); /* 1, 0.5 */
239 
240  /* [dx+aw,dz+ay,dy-az,dw-ax]+-[bz,bx,bw,by] = [dx+aw+bz,dz+ay-bx,dy-az+bw,dw-ax-by]*/
241  e = _mm_addsub_ps (e, m2); /* 3, 1 */
242 
243  /* 2 */
244  /* [dz+ay-bx,dw-ax-by,dy-az+bw,dx+aw+bz] */
245  e = _mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(e), _MM_SHUFFLE (2, 0, 1, 3))); /* 1, 0.5 */
246 
247  /* [dz+ay-bx,dw-ax-by,dy-az+bw,dx+aw+bz]+-[cw,cz,cx,cy]
248  = [dz+ay-bx+cw,dw-ax-by-cz,dy-az+bw+cx,dx+aw+bz-cy] */
249  e = _mm_addsub_ps (e, m3); /* 3, 1 */
250 
251  /* 2 */
252  /* [dw-ax-by-cz,dz+ay-bx+cw,dy-az+bw+cx,dx+aw+bz-cy] */
253  quaternion tmp;
254  reinterpret_cast<vectorSIMDf&>(tmp) = vectorSIMDf(_mm_castsi128_ps(_mm_shuffle_epi32(_mm_castps_si128(e), _MM_SHUFFLE (2, 3, 1, 0)))); /* 1, 0.5 */
255  return tmp;
256  }

407 inline void quaternion::getMatrix(matrix4x3 &dest,
408  const core::vector3df_SIMD &center) const
409 {
410  dest(0,0) = 1.0f - 2.0f*Y*Y - 2.0f*Z*Z;
411  dest(1,0) = 2.0f*X*Y + 2.0f*Z*W;
412  dest(2,0) = 2.0f*X*Z - 2.0f*Y*W;
413 
414  dest(0,1) = 2.0f*X*Y - 2.0f*Z*W;
415  dest(1,1) = 1.0f - 2.0f*X*X - 2.0f*Z*Z;
416  dest(2,1) = 2.0f*Z*Y + 2.0f*X*W;
417 
418  dest(0,2) = 2.0f*X*Z + 2.0f*Y*W;
419  dest(1,2) = 2.0f*Z*Y - 2.0f*X*W;
420  dest(2,2) = 1.0f - 2.0f*X*X - 2.0f*Y*Y;
421 
422  dest(0,3) = center.X;
423  dest(1,3) = center.Y;
424  dest(2,3) = center.Z;

478 // set this quaternion to the result of the interpolation between two quaternions
479 inline quaternion quaternion::slerp(quaternion q1, const quaternion &q2, const float& interpolant, const float& threshold)
480 {
481  vectorSIMDf angle = q1.dotProduct(q2);
482 
483  // make sure we use the short rotation
484  if (angle.X < 0.0f)
485  {
486  q1 *= -1.0f;
487  angle *= -1.0f;
488  }
489 
490  if (angle.X <= (1.f-threshold)) // spherical interpolation
491  {
492  const f32 theta = acosf(angle.X);
493  const vectorSIMDf sintheta(sinf(theta));
494  const vectorSIMDf scale(sinf(theta * (1.0f-interpolant)));
495  const vectorSIMDf invscale(sinf(theta * interpolant));
496  vectorSIMDf retval = (reinterpret_cast<vectorSIMDf&>(q1)*scale + reinterpret_cast<const vectorSIMDf&>(q2)*invscale)/sintheta;
497  return reinterpret_cast<quaternion&>(retval);
498  }
499  else // linear interploation
500  return lerp(q1,q2,interpolant);
501 }

536 inline void quaternion::toEuler(vector3df_SIMD& euler) const
537 {
538  const f64 sqw = W*W;
539  const f64 sqx = X*X;
540  const f64 sqy = Y*Y;
541  const f64 sqz = Z*Z;
542  vectorSIMDf sqr = *reinterpret_cast<const vectorSIMDf*>(this);
543  sqr *= sqr;
544  const f64 test = 2.0 * (Y*W - X*Z);
545 
546  if (core::equals(test, 1.0, 0.000001))
547  {
548  // heading = rotation about z-axis
549  euler.Z = (f32) (-2.0*atan2(X, W));
550  // bank = rotation about x-axis
551  euler.X = 0;
552  // attitude = rotation about y-axis
553  euler.Y = (f32) (PI64/2.0);
554  }
555  else if (core::equals(test, -1.0, 0.000001))
556  {
557  // heading = rotation about z-axis
558  euler.Z = (f32) (2.0*atan2(X, W));
559  // bank = rotation about x-axis
560  euler.X = 0;
561  // attitude = rotation about y-axis
562  euler.Y = (f32) (PI64/-2.0);
563  }
564  else
565  {
566  // heading = rotation about z-axis
567  euler.Z = (f32) atan2(2.0 * (X*Y +Z*W),(sqr.X - sqr.Y - sqr.Z + sqr.W));
568  // bank = rotation about x-axis
569  euler.X = (f32) atan2(2.0 * (Y*Z +X*W),(-sqr.X - sqr.Y + sqr.Z + sqr.W));
570  // attitude = rotation about y-axis
571  euler.Y = (f32) asin( core::clamp(test, -1.0, 1.0) );
572  }
573 }

575 inline quaternion quaternion::rotationFromTo(const vector3df_SIMD& from, const vector3df_SIMD& to)
576 {
577  // Based on Stan Melax's article in Game Programming Gems
578  // Copy, since cannot modify local
579  vector3df_SIMD v0 = from;
580  vector3df_SIMD v1 = to;
581  v0 = core::normalize(v0);
582  v1 = core::normalize(v1);
583 
584  const vectorSIMDf dddd = v0.dotProduct(v1);
585  quaternion tmp;
586  if (dddd.X >= 1.0f) // If dot == 1, vectors are the same
587  {
588  return tmp;
589  }
590  else if (dddd.X <= -1.0f) // exactly opposite
591  {
592  vector3df_SIMD axis(1.0f, 0.f, 0.f);
593  axis = axis.crossProduct(v0);
594  if (axis.getLengthAsFloat()==0.f)
595  {
596  axis.set(0.f,1.f,0.f);
597  axis = axis.crossProduct(v0);
598  }
599  // same as fromAngleAxis(PI, axis).normalize();
600  reinterpret_cast<vectorSIMDf&>(tmp) = axis;
601  return normalize(tmp);
602  }
603 
604  vectorSIMDf s = sqrt(vectorSIMDf(2.f,2.f,2.f,0.f)+dddd*2.f);
605  reinterpret_cast<vectorSIMDf&>(tmp) = v0.crossProduct(v1)*reciprocal(s);
606  tmp.W = s.X*0.5f;
607  return normalize(tmp);
