518  inline void rotateXZByRAD(const float &radians, const vectorSIMDf& center)
519  {
520  __m128 xmm1 = center.getAsRegister();
521  __m128 xmm0 = _mm_sub_ps(getAsRegister(),xmm1);
522 
523  float cs = cosf(radians);
524  float sn = sinf(radians);

613  between two scene nodes, then applying the result of getHorizontalAngle() to one scene node will point

615  Example code:
616  // Where target and seeker are of type ISceneNode*

622  +Z (e.g. 0, 0, 1) direction vector would make it point in the same direction as this vector. The Z (roll) rotation
623  is always 0, since two Euler rotations are sufficient to point in any given direction. *
624  inline vectorSIMDf getHorizontalAngle3D() const
625  {

627 
628  const float tmp = atan2f(x,z);

632  xmm0 = _mm_add_ps(xmm0,FAST_FLOAT_SHUFFLE(xmm0,_MM_SHUFFLE(3,0,1,2)));
