384  inline bool getInverse(matrix4x3& out) const
385  {
389  f32 d = column[0].dotProduct(column[1].crossProduct(column[2]));
390 
391  if( core::iszero ( d, FLT_MIN ) )
392  return false;
393 
394  vectorSIMDf rows[3];
395  rows[0].set(column[1].crossProduct(column[2]));
396  rows[1].set(column[2].crossProduct(column[0]));
397  rows[2].set(column[0].crossProduct(column[1]));
398 
399  rows[0] /= d;
400  rows[1] /= d;
401  rows[2] /= d;
402  //transpose
403  out.column[0].set(rows[0].X,rows[1].X,rows[2].X);
404  out.column[1].set(rows[0].Y,rows[1].Y,rows[2].Y);
405  out.column[2].set(rows[0].Z,rows[1].Z,rows[2].Z);
406 
407  //out.column[3] = out.column[0]*m.column[3].X+out.column[1]*m.column[3].Y+out.column[2]*m.column[3].Z;
408  out.mulSub3x3With3x1(reinterpret_cast<float*>(out.column+3),reinterpret_cast<const float*>(column+3));
409  out.column[3] = -out.column[3];
410 
411  return true;
412  }

615  inline core::vector3df matrix4x3::getRotationDegrees() const
616  {
617  const matrix4x3 &mat = *this;
618  core::vector3df scale = getScale();
619  // we need to check for negative scale on to axes, which would bring up wrong results
620  if (scale.Y<0 && scale.Z<0)
621  {
622  scale.Y =-scale.Y;
623  scale.Z =-scale.Z;
624  }
625  else if (scale.X<0 && scale.Z<0)
626  {
627  scale.X =-scale.X;
628  scale.Z =-scale.Z;
629  }
630  else if (scale.X<0 && scale.Y<0)
631  {
632  scale.X =-scale.X;
633  scale.Y =-scale.Y;
634  }
635  const core::vector3d<float> invScale(core::reciprocal(scale.X),core::reciprocal(scale.Y),core::reciprocal(scale.Z));
636 
637  float Y = -asinf(core::clamp(mat(2,0)*invScale.X, -1.f, 1.f));
638  const float C = cosf(Y);
639  Y *= RADTODEG64;
640 
641  float rotx, roty, X, Z;
642 
643  if (!core::iszero(C))
644  {
645  const float invC = core::reciprocal(C);
646  rotx = mat(2,2) * invC * invScale.Z;
647  roty = mat(2,1) * invC * invScale.Y;
648  X = atan2f( roty, rotx ) * RADTODEG64;
649  rotx = mat(0,0) * invC * invScale.X;
650  roty = mat(1,0) * invC * invScale.X;
651  Z = atan2f( roty, rotx ) * RADTODEG64;
652  }
653  else
654  {
655  X = 0.0;
656  rotx = mat(1,1) * invScale.Y;
657  roty = -mat(0,1) * invScale.Y;
658  Z = atan2f( roty, rotx ) * RADTODEG64;
659  }
660 
661  // fix values that get below zero
662  if (X < 0.0) X += 360.0;
663  if (Y < 0.0) Y += 360.0;
664  if (Z < 0.0) Z += 360.0;
665 
666  return vector3df((float)X,(float)Y,(float)Z);
667  }

817  inline matrix4x3& matrix4x3::buildRotateFromTo(const core::vector3df& from, const core::vector3df& to)
818  {
819  // unit vectors
820  core::vector3df f(from);
821  core::vector3df t(to);
822  f.normalize();
823  t.normalize();
824 
825  // axis multiplication by sin
826  core::vector3df vs(t.crossProduct(f));
827 
828  // axis of rotation
829  core::vector3df v(vs);
830  v.normalize();
831 
832  // cosinus angle
833  float ca = f.dotProduct(t);
834 
835  core::vector3df vt(v * (1 - ca));
836 
837  column[0].X = vt.X * v.X + ca;
838  column[1].Y = vt.Y * v.Y + ca;
839  column[2].Z = vt.Z * v.Z + ca;
840 
841  vt.X *= v.Y;
842  vt.Z *= v.X;
843  vt.Y *= v.Z;
844 
845  column[0].Y = vt.X - vs.Z;
846  column[0].Z = vt.Z + vs.Y;
847 
848  column[1].X = vt.X + vs.Z;
849  column[1].Z = vt.Y - vs.X;
850 
851  column[2].X = vt.Z - vs.Y;
852  column[2].Y = vt.Y + vs.X;
853 
854  column[3].set(0.f,0.f,0.f);
855 
856  return *this;
857  }

867  inline void matrix4x3::buildAxisAlignedBillboard(

869  const core::vector3df& center,

905
