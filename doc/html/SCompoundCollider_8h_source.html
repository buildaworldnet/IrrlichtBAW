173  inline bool CollideWithRay(float& collisionDistance, vectorSIMDf origin, vectorSIMDf direction, const float& dirMaxMultiplier) const
174  {
175  if (colliderData.attachedNode)
176  {
177  matrix4x3 absoluteTransform = colliderData.attachedNode->getAbsoluteTransformation();
178  bool retval = absoluteTransform.makeInverse();
179  if (!retval)
180  return false;
181 
182  absoluteTransform.transformVect(origin.pointer,origin.pointer);
183  origin.pointer[3] = 0.f;
184  absoluteTransform.mulSub3x3With3x1(direction.pointer,direction.pointer);
185 
186  switch (colliderData.attachedNode->getType())
187  {
188  case scene::ESNT_MESH_INSTANCED:
189  {
190  core::matrix4x3 instanceTform = static_cast<scene::IMeshSceneNodeInstanced*>(colliderData.attachedNode)->getInstanceTransform(colliderData.instanceID);
191  retval = instanceTform.makeInverse();
192  if (!retval)
193  return false;
194 
195  instanceTform.transformVect(origin.pointer,origin.pointer);
196  origin.pointer[3] = 0.f;
197  instanceTform.mulSub3x3With3x1(direction.pointer,direction.pointer);
198  }
199  break;
201  default:
202  break;
203  }
204  }
205 
206  vectorSIMDf direction_reciprocal = reciprocal(direction);
207  float dummyPosition;
208  if (!BBox.CollideWithRay(dummyPosition,origin,direction,dirMaxMultiplier,direction_reciprocal))
209  {
210  return false;
211  }
212 
213 
214  for (size_t i=0; i<Shapes.size(); i++)
215  {
216  switch (Shapes[i].objectType)
217  {
218  case SCollisionShapeDef::ECST_AABOX:
219  {
220  SAABoxCollider* tmp = static_cast<SAABoxCollider*>(Shapes[i].object);
221  if (tmp->CollideWithRay(collisionDistance,origin,direction,dirMaxMultiplier,direction_reciprocal))
222  return true;
223  }
224  break;
225  case SCollisionShapeDef::ECST_ELLIPSOID:
226  {
227  SEllipsoidCollider* tmp = static_cast<SEllipsoidCollider*>(Shapes[i].object);
228  if (tmp->CollideWithRay(collisionDistance,origin,direction,dirMaxMultiplier))
229  return true;
