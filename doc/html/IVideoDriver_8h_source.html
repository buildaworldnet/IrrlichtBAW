136  void apply(SMaterial& material)
137  {
138  if (Enabled)
139  {
140  for (u32 i=0; i<32; ++i)
141  {
142  const u32 num=(1<<i);
143  if (EnableFlags & num)
144  {
145  switch (num)
146  {
147  case EMF_WIREFRAME: material.Wireframe = Material.Wireframe; break;
148  case EMF_POINTCLOUD: material.PointCloud = Material.PointCloud; break;
149  case EMF_ZBUFFER: material.ZBuffer = Material.ZBuffer; break;
150  case EMF_ZWRITE_ENABLE: material.ZWriteEnable = Material.ZWriteEnable; break;
151  case EMF_BACK_FACE_CULLING: material.BackfaceCulling = Material.BackfaceCulling; break;
152  case EMF_FRONT_FACE_CULLING: material.FrontfaceCulling = Material.FrontfaceCulling; break;
153  case EMF_ANTI_ALIASING: material.AntiAliasing = Material.AntiAliasing; break;
154  case EMF_COLOR_MASK: material.ColorMask = Material.ColorMask; break;
155  case EMF_BLEND_OPERATION: material.BlendOperation = Material.BlendOperation; break;
156  }
157  }
158  }
159  }
160  }
